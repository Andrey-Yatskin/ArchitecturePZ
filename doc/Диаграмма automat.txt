@startuml
(ERROR)
(Off)
(On)
(Waiting)
(Move_forward)
(turn_left)
(Cleaning)

(Off) -> (On) : press_start
(On) -down-> (ERROR)
(Cleaning) -down-> (ERROR)
(Move_forward) -down-> (ERROR)
(turn_left) -down-> (ERROR)
(On) -up-> (Waiting) : waiting_command
(Waiting) -down-> (Move_forward) : receive_move
(Waiting) -left-> (Off) : press_stop
(Move_forward) -right-> (turn_left) : receive_turn
(turn_left) -right -> (Cleaning) : receive_clean

@enduml
